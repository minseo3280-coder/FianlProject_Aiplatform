final_project_aiplatform/
├── .venv/                    # [통합 가상환경] 모든 하위 폴더가 이 환경을 공통으로 사용
├── .gitignore                # 루트 .venv 및 데이터 파일 제외 설정
├── requirements.txt          # [통합 라이브러리 목록] (FastAPI + YOLO + Camera 등 전부 포함)
├── README.md                 # 전체 AI 프로젝트 가이드
│
├── backend-ai/               # [FastAPI] AI 서빙 서버
│   ├── app/                  # main.py, routers
│   ├── core/                 # AI 모델 로드 클래스
│   └── README.md             # API 실행법
│
├── ai-engine/                # [Model Training] 모델 학습 및 연구
│   ├── data/                 # 데이터셋 (Git 제외)
│   ├── models/               # 학습된 모델 파일 (Git 제외)
│   ├── notebooks/            # 주피터 노트북
│   └── train.py              # 학습 스크립트
│
└── camera/                   # [Device] 카메라 제어 및 스트리밍
    ├── capture.py            # 카메라 영상 캡처 스크립트
    └── stream.py             # 영상 송출 로직

-- 버전 설치 --
1. aiplatform 폴더 내에 venv 생성 python -m venv .venv 
2. venv activate  .\.venv\Scripts\activate
3. pip install -r requirements.txt 터미널에서 실행
